\hypertarget{index_autotoc_md1}{}\doxysection{Get Started}\label{index_autotoc_md1}
Requires CMake $>$ 3.\+13.\+0 and a C++ compiler with C++11 support\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{git clone -\/-\/recursive git@github.com:Pressio/pressio-\/tutorials.git}
\DoxyCodeLine{export CXX=<path-\/to-\/your-\/CXX-\/compiler>}
\DoxyCodeLine{cd pressio-\/tutorials \&\& mkdir build \&\& cd build}
\DoxyCodeLine{cmake -\/DCMAKE\_BUILD\_TYPE=Release ..}
\DoxyCodeLine{make -\/j4}

\end{DoxyCode}
